async function fetchPlanilha() { const agora = new Date(); const hojeCompleto = `${String(agora.getDate()).padStart(2, '0')}/${String(agora.getMonth() + 1).padStart(2, '0')}/${agora.getFullYear()}`; try { const response = await fetch(`${spreadsheetUrl}×tamp=${new Date().getTime()}`); if (!response.ok) throw new Error("Erro na resposta"); const csv = await response.text(); const linhas = csv.split(/\r?\n/); if (linhas.length === 0) return; const separador = linhas[0].includes(';') ? ';' : ','; // Cabeçalho: Limpa apenas as aspas e espaços nas extremidades para manter o nome exato const cabecalho = linhas[0].split(separador).map(txt => txt.replace(/"/g, '').trim() ); // Localização exata dos índices const idxData = cabecalho.indexOf("DATA"); // Nome exato conforme sua solicitação const nomeColunaAlvo = "TOTAL MOTORISTA DISTRIBUIÇÃO (QTD)"; const idxMdTotal = cabecalho.indexOf(nomeColunaAlvo); // Mapeamento dos demais campos (mantendo os nomes originais) const mapping = { vol: "VOLUME EXPEDIDO (KG)", drop: "DROPSIZE (KG)", ocp: "OCUPAÇÃO (%)", nf: "NF'S (QTD)", fTotal: "FROTA PRÓPRIA ENTREGA TOTAL (QTD)", fParados: "FROTA PRÓPRIA ENTREGA PARADOS (QTD)", fDispo: "FROTA PRÓPRIA ENTREGA DISPONIBILIDADE (QTD)", fUtil: "FROTA PRÓPRIA ENTREGA UTILIZADOS (QTD)", ftTotal: "FROTA PRÓPRIA TRANSF. TOTAL (QTD)", ftParados: "FROTA PRÓPRIA TRANSF. PARADOS (QTD)", ftDispo: "FROTA PRÓPRIA TRANSF. TOTAL DISPONIBILIDADE (QTD)", ftUtil: "FROTA PRÓPRIA TRANSF. UTILIZADOS (QTD)", terceiros: "TERCEIROS UTILIZADOS (QTD)", mdVagas: "VAGAS EM ABERTO MOTORISTA DISTRIBUIÇÃO (QTD)", mdFerias: "FÉRIAS MOTORISTA DISTRIBUIÇÃO (QTD)", mdFaltas: "FALTAS MOTORISTA DISTRIBUIÇÃO (QTD)", mdAtestados: "ATESTADOS MOTORISTA DISTRIBUIÇÃO (QTD)", mdBanco: "BANCO DE HORAS MOTORISTA DISTRIBUIÇÃO (QTD)", mdUtil: "UTILIZADOS MOTORISTA DISTRIBUIÇÃO (QTD)" }; let achouData = false; let v = { mdTotal: "-" }; // Inicializa o valor alvo for(let k in mapping) v[k] = "-"; if (idxData !== -1) { for (let i = 1; i < linhas.length; i++) { const colunas = linhas[i].split(separador); let dataPlanilha = colunas[idxData] ? colunas[idxData].replace(/"/g, '').trim() : ""; if (dataPlanilha.includes(hojeCompleto)) { achouData = true; const limpar = (idx) => { if (idx === -1 || idx === undefined || colunas[idx] === undefined) return "-"; let val = colunas[idx].replace(/"/g, '').trim(); return (val === "" || val === "null" || val === "#N/A") ? "-" : val; }; // 1. Aplica a lógica específica para o TOTAL MOTORISTA DISTRIBUIÇÃO (QTD) v.mdTotal = limpar(idxMdTotal); // 2. Preenche os outros campos for(let key in mapping) { const idxOutro = cabecalho.indexOf(mapping[key]); v[key] = limpar(idxOutro); } break; } } } // Atualização dos elementos na tela document.getElementById('vol-valor').textContent = v.vol; document.getElementById('drop-valor').textContent = v.drop; document.getElementById('ocp-valor').textContent = v.ocp; document.getElementById('nf-valor').textContent = v.nf; document.getElementById('f-total').textContent = v.fTotal; document.getElementById('f-parados').textContent = v.fParados; document.getElementById('f-dispo').textContent = v.fDispo; document.getElementById('f-util').textContent = v.fUtil; document.getElementById('ft-total').textContent = v.ftTotal; document.getElementById('ft-parados').textContent = v.ftParados; document.getElementById('ft-dispo').textContent = v.ftDispo; document.getElementById('ft-util').textContent = v.ftUtil; document.getElementById('terc-util').textContent = v.terceiros; // O ID md-total recebe o valor da coluna TOTAL MOTORISTA DISTRIBUIÇÃO (QTD) document.getElementById('md-total').textContent = v.mdTotal; document.getElementById('md-vagas').textContent = v.mdVagas; document.getElementById('md-ferias').textContent = v.mdFerias; document.getElementById('md-faltas').textContent = v.mdFaltas; document.getElementById('md-atestados').textContent = v.mdAtestados; document.getElementById('md-banco').textContent = v.mdBanco; document.getElementById('md-util').textContent = v.mdUtil; const horaSinc = new Date().toLocaleTimeString('pt-BR', {hour:'2-digit', minute:'2-digit', second:'2-digit'}); const statusIds = ['status-vol', 'status-drop', 'status-ocp', 'status-nf', 'status-frota', 'status-frota-t', 'status-terceiros', 'status-md']; statusIds.forEach(id => { const el = document.getElementById(id); if (el) { if (achouData) { el.textContent = `Sincronizado: ${horaSinc}`; el.style.color = "#00ced1"; } else { el.textContent = `Aguardando dado de hoje (${hojeCompleto})`; el.style.color = "orange"; } } }); } catch (err) { console.error("Erro ao buscar dados:", err); } }
